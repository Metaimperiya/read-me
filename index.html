// app.js — главный контроллер

const state = {
    user: null,          // текущий пользователь
    uid: null,           // UID
    currentTab: 'feed',  // feed | groups | people
    currentRoom: null,   // ID комнаты
    chatPartner: null,   // UID собеседника
    isAdmin: false,
    notifications: [],
    privacy: { hideFriends: false }
};

// Роутинг (без перезагрузки)
function navigateTo(page, params) {
    // page: 'feed' | 'groups' | 'people' | 'profile' | 'chat'
    // params: { uid, roomId, chatId }
    
    document.querySelectorAll('.page').forEach(p => p.style.display = 'none');
    
    const section = document.getElementById(page + 'Page');
    if (section) section.style.display = 'block';
    
    if (page === 'feed') loadFeed();
    if (page === 'groups') loadRooms();
    if (page === 'people') loadPeople();
    if (page === 'profile') renderProfile(params.uid);
    if (page === 'chat') openChat(params);
}

📄 sw.js — Service Worker

  • Кеширование — загружает сайт в кеш при первом открытии
  • Офлайн-режим — сайт работает без интернета
  • Быстрая загрузка — повторные открытия мгновенные

Регистрация в index.html:

if ('serviceWorker' in navigator) {
    navigator.serviceWorker.register('/sw.js');
}
